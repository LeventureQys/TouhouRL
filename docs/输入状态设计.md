# 输入状态设计详解

## 问题：为什么不能直接用原始像素？

直接使用游戏画面的原始像素（如640x480x3）作为输入会有以下问题：

1. **维度过高**：640×480×3 = 921,600个输入，网络参数量巨大
2. **训练困难**：高维输入导致训练极慢，容易过拟合
3. **信息冗余**：背景、特效等无关信息占据大量空间
4. **计算开销**：每帧处理时间长，无法实时决策

## 解决方案：三种输入设计方案

### 方案A：特征向量输入（推荐初期使用）

**核心思想**：通过计算机视觉提取关键信息，转换为结构化特征向量

#### 状态表示

```python
state = {
    # 自机信息 (4维)
    'player_x': float,          # 归一化到[0,1]

    'player_vx': float,         # 速度x

    'lives': int,               # 残机数
    'bombs': int,               # Bomb数

        (x, y, type),           # 位置、类型
        ...
    ],

    'danger_8dir': [            # 8个方向的危险度
        danger_up,
        danger_down,
        danger_left,
```

#### 特征向量构建

```python
def build_feature_vector(game_state):
    """构建特征向量"""
    features = []

    # 1. 自机信息 (4维)
    features.extend([
        game_state['player_x'] / SCREEN_WIDTH,   # 归一化
        game_state['player_y'] / SCREEN_HEIGHT,
        game_state['player_vx'] / MAX_VELOCITY,
        game_state['player_vy'] / MAX_VELOCITY
    ])

    # 2. 生命信息 (3维)
    features.extend([
        game_state['lives'] / MAX_LIVES,
        game_state['bombs'] / MAX_BOMBS,
        game_state['power_level'] / MAX_POWER
    ])

    # 3. 最近的20个子弹 (20×5=100维)
    bullets = game_state['bullets'][:20]  # 只取最近的20个
    for bullet in bullets:
        features.extend([
            bullet['x'] / SCREEN_WIDTH,
            bullet['y'] / SCREEN_HEIGHT,
            bullet['vx'] / MAX_BULLET_VELOCITY,
            bullet['vy'] / MAX_BULLET_VELOCITY,
            bullet['type'] / NUM_BULLET_TYPES  # 类型编码
        ])

    # 如果子弹不足20个，用0填充
    for _ in range(20 - len(bullets)):
        features.extend([0, 0, 0, 0, 0])

    # 4. 最近的5个敌机 (5×4=20维)
    enemies = game_state['enemies'][:5]
    for enemy in enemies:
        features.extend([
            enemy['x'] / SCREEN_WIDTH,
            enemy['y'] / SCREEN_HEIGHT,
            enemy['hp'] / MAX_ENEMY_HP,
            enemy['type'] / NUM_ENEMY_TYPES
        ])

    # 填充
    for _ in range(5 - len(enemies)):
        features.extend([0, 0, 0, 0])

    # 5. 最近的10个道具 (10×3=30维)
    items = game_state['items'][:10]
    for item in items:
        features.extend([
            item['x'] / SCREEN_WIDTH,
            item['y'] / SCREEN_HEIGHT,
            item['type'] / NUM_ITEM_TYPES
        ])

    # 填充
    for _ in range(10 - len(items)):
        features.extend([0, 0, 0])

    # 6. 8方向危险度 (8维)
    features.extend(game_state['danger_8dir'])

    # 总维度：4 + 3 + 100 + 20 + 30 + 8 = 165维
    return np.array(features, dtype=np.float32)
```

#### 网络架构

```python
import torch
import torch.nn as nn

class DQN_FeatureVector(nn.Module):
    def __init__(self, state_dim=165, action_dim=36):
        super().__init__()

        self.network = nn.Sequential(
            nn.Linear(state_dim, 256),
            nn.ReLU(),
            nn.Linear(256, 256),
            nn.ReLU(),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, action_dim)
        )

    def forward(self, state):
        return self.network(state)
```

**优点**：
- 维度低（约165维），训练快
- 信息密集，无冗余
- 易于调试和理解

**缺点**：
- 依赖准确的目标检测
- 可能丢失一些视觉细节
- 需要手动设计特征

### 方案B：简化图像输入

**核心思想**：降低图像分辨率，只保留关键信息

#### 图像预处理

```python
def preprocess_frame(frame):
    """预处理游戏画面"""
    # 1. 裁剪游戏区域（去掉UI）
    game_area = frame[0:480, 0:384]  # 假设游戏区域是384x480

    # 2. 转灰度
    gray = cv2.cvtColor(game_area, cv2.COLOR_BGR2GRAY)

    # 3. 降采样到84x84（Atari DQN的标准尺寸）
    resized = cv2.resize(gray, (84, 84), interpolation=cv2.INTER_AREA)

    # 4. 归一化到[0,1]
    normalized = resized / 255.0

    return normalized
```

#### 帧堆叠（Frame Stacking）

```python
class FrameStack:
    def __init__(self, num_frames=4):
        self.num_frames = num_frames
        self.frames = []

    def reset(self):
        self.frames = []

    def add_frame(self, frame):
        self.frames.append(frame)
        if len(self.frames) > self.num_frames:
            self.frames.pop(0)

    def get_state(self):
        # 如果帧数不足，重复最后一帧
        while len(self.frames) < self.num_frames:
            if len(self.frames) == 0:
                self.frames.append(np.zeros((84, 84)))
            else:
                self.frames.append(self.frames[-1])

        # 堆叠成 (4, 84, 84)
        return np.stack(self.frames, axis=0)
```

#### 网络架构（CNN）

```python
class DQN_CNN(nn.Module):
    def __init__(self, action_dim=36):
        super().__init__()

        # 输入：(4, 84, 84) - 4帧灰度图
        self.conv = nn.Sequential(
            # 第一层卷积
            nn.Conv2d(4, 32, kernel_size=8, stride=4),  # -> (32, 20, 20)
            nn.ReLU(),

            # 第二层卷积
            nn.Conv2d(32, 64, kernel_size=4, stride=2), # -> (64, 9, 9)
            nn.ReLU(),

            # 第三层卷积
            nn.Conv2d(64, 64, kernel_size=3, stride=1), # -> (64, 7, 7)
            nn.ReLU()
        )

        # 全连接层
        self.fc = nn.Sequential(
            nn.Linear(64 * 7 * 7, 512),
            nn.ReLU(),
            nn.Linear(512, action_dim)
        )

    def forward(self, state):
        # state: (batch, 4, 84, 84)
        conv_out = self.conv(state)
        conv_out = conv_out.view(conv_out.size(0), -1)  # 展平
        return self.fc(conv_out)
```

**优点**：
- 不需要目标检测
- 保留视觉信息
- 经典方法，有大量参考

**缺点**：
- 仍然较慢（相比特征向量）
- 需要更多训练样本
- 84×84×4 = 28,224维输入

### 方案C：混合输入（高级方案）

**核心思想**：结合图像特征和结构化特征

#### 状态表示

```python
state = {

    'features': np.array    # 关键特征向量 (20维)

    player_x, player_y,     # 自机位置

    nearest_bullet_angle,   # 最近子弹角度
    danger_level,           # 整体危险度

    item_count,             # 道具数量
    # ... 其他关键信息
]
```

#### 网络架构（双流网络）

```python
class DQN_Hybrid(nn.Module):
    def __init__(self, feature_dim=20, action_dim=36):
        super().__init__()

        # 图像流（CNN）
        self.image_stream = nn.Sequential(
            nn.Conv2d(4, 32, kernel_size=8, stride=4),
            nn.ReLU(),
            nn.Conv2d(32, 64, kernel_size=4, stride=2),
            nn.ReLU(),
            nn.Conv2d(64, 64, kernel_size=3, stride=1),
            nn.ReLU()
        )

        # 特征流（MLP）
        self.feature_stream = nn.Sequential(
            nn.Linear(feature_dim, 64),
            nn.ReLU(),
            nn.Linear(64, 64),
            nn.ReLU()
        )

        # 融合层
        self.fusion = nn.Sequential(
            nn.Linear(64 * 7 * 7 + 64, 512),
            nn.ReLU(),
            nn.Linear(512, action_dim)
        )

    def forward(self, image, features):
        # 图像特征
        img_out = self.image_stream(image)
        img_out = img_out.view(img_out.size(0), -1)

        # 结构化特征
        feat_out = self.feature_stream(features)

        # 拼接
        combined = torch.cat([img_out, feat_out], dim=1)

        return self.fusion(combined)
```

**优点**：
- 结合两种方法的优势
- 性能最好

**缺点**：
- 最复杂
- 训练最慢

## 推荐的渐进式方案

### 阶段1：特征向量（1-2周）

使用**方案A**快速验证：
- 实现目标检测
- 构建特征向量
- 训练简单的MLP网络
- 目标：在简单场景中存活

```python

    danger_8dir                   # 8维

```

### 阶段2：完整特征向量（2-3周）

扩展到完整的特征向量：
- 加入敌机信息
- 加入道具信息
- 更多子弹信息
- 目标：击破第一关Boss

### 阶段3：图像输入（可选，3-4周）

如果特征向量效果不够好，尝试**方案B**：
- 实现图像预处理
- 训练CNN网络
- 可能需要更多训练时间

### 阶段4：混合输入（高级，4周+）

追求最佳性能时使用**方案C**

## 实现示例：完整的特征向量环境

```python
import numpy as np
import gym
from gym import spaces

class TouhouEnv(gym.Env):
    def __init__(self):
        super().__init__()

        # 动作空间
        self.action_space = spaces.Discrete(36)

        # 状态空间：165维特征向量
        self.observation_space = spaces.Box(
            low=-1.0, high=1.0,
            shape=(165,),
            dtype=np.float32
        )

        self.vision_system = VisionSystem()  # 视觉检测系统

    def _get_observation(self):
        """获取当前状态"""
        # 1. 捕获屏幕
        frame = capture_screen()

        # 2. 目标检测
        detections = self.vision_system.detect(frame)

        # 3. 构建游戏状态
        game_state = {
            'player_x': detections['player']['x'],
            'player_y': detections['player']['y'],
            'player_vx': detections['player']['vx'],
            'player_vy': detections['player']['vy'],
            'lives': detections['lives'],
            'bombs': detections['bombs'],
            'power_level': detections['power'],
            'bullets': detections['bullets'],
            'enemies': detections['enemies'],
            'items': detections['items'],
            'danger_8dir': self._calculate_danger(detections)
        }

        # 4. 转换为特征向量
        feature_vector = build_feature_vector(game_state)

        return feature_vector

    def _calculate_danger(self, detections):
        """计算8个方向的危险度"""
        player_pos = np.array([detections['player']['x'],
                               detections['player']['y']])

        # 8个方向
        directions = [
            (0, -1),   # 上
            (0, 1),    # 下
            (-1, 0),   # 左
            (1, 0),    # 右
            (-1, -1),  # 左上
            (1, -1),   # 右上
            (-1, 1),   # 左下
            (1, 1)     # 右下
        ]

        danger_scores = []
        for dx, dy in directions:
            # 计算该方向的危险度
            danger = 0
            test_pos = player_pos + np.array([dx, dy]) * 10  # 测试位置

            for bullet in detections['bullets']:
                bullet_pos = np.array([bullet['x'], bullet['y']])
                dist = np.linalg.norm(test_pos - bullet_pos)

                # 距离越近，危险度越高
                if dist < 50:
                    danger += 1.0 / (dist + 1)

            danger_scores.append(min(danger, 1.0))  # 归一化到[0,1]

        return danger_scores

    def step(self, action):
        # 执行动作
        self._execute_action(action)

        # 获取新状态
        observation = self._get_observation()

        # 计算奖励
        reward = self._calculate_reward()

        # 检查结束
        done = self._is_done()

        info = {}

        return observation, reward, done, info
```

## 总结

**推荐路线**：
1. 从**方案A（特征向量）**开始
2. 先用简化版（50维左右）快速验证
3. 逐步扩展到完整版（165维）
4. 如果效果不够好，再考虑方案B或C

**关键点**：
- 不要直接用原始像素！
- 特征向量是最实用的方案
- 归一化所有输入到[-1, 1]或[0, 1]
- 对于数量不定的对象（子弹、敌机），取最近的K个并填充
