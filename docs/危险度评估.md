# 危险度评估系统设计

## 问题分析

危险度评估需要考虑：
1. **子弹位置**：距离自机的远近
2. **子弹速度方向**：是否朝向自机
3. **子弹大小**：判定半径
4. **未来轨迹**：预测子弹未来位置
5. **多个子弹的综合威胁**：密集弹幕的危险度更高

## 危险度评估方法

### 方法1：基于轨迹预测的危险度

**核心思想**：预测子弹未来N帧的轨迹，计算与自机可能位置的最小距离

#### 单个子弹的危险度计算

```python
def calculate_bullet_danger(bullet, player_pos, player_velocity, time_horizon=30):
    """
    计算单个子弹对自机的危险度

    Args:
        bullet: 子弹信息 {x, y, vx, vy, radius, type}
        player_pos: 自机当前位置 (x, y)
        player_velocity: 自机当前速度 (vx, vy)
        time_horizon: 预测时间范围（帧数）

    Returns:
        danger_score: 危险度 [0, 1]
    """
    danger_score = 0.0

    # 1. 预测子弹和自机的未来轨迹
    min_distance = float('inf')
    collision_time = None

    for t in range(time_horizon):
        # 子弹未来位置（考虑加速度）
        bullet_future_x = bullet['x'] + bullet['vx'] * t
        bullet_future_y = bullet['y'] + bullet['vy'] * t

        # 如果子弹有加速度
        if 'ax' in bullet and 'ay' in bullet:
            bullet_future_x += 0.5 * bullet['ax'] * t * t
            bullet_future_y += 0.5 * bullet['ay'] * t * t

        # 自机未来位置（假设保持当前速度）
        player_future_x = player_pos[0] + player_velocity[0] * t
        player_future_y = player_pos[1] + player_velocity[1] * t

        # 计算距离
        dx = bullet_future_x - player_future_x
        dy = bullet_future_y - player_future_y
        distance = np.sqrt(dx * dx + dy * dy)

        # 考虑子弹和自机的判定半径
        effective_distance = distance - bullet['radius'] - PLAYER_HITBOX_RADIUS

        if effective_distance < min_distance:
            min_distance = effective_distance
            collision_time = t

    # 2. 根据最小距离计算危险度
    if min_distance < 0:
        # 会碰撞
        danger_score = 1.0
    elif min_distance < 50:
        # 距离越近，危险度越高
        danger_score = 1.0 - (min_distance / 50.0)
    else:
        danger_score = 0.0

    # 3. 考虑时间因素：越早碰撞越危险
    if collision_time is not None and collision_time < time_horizon:
        time_factor = 1.0 - (collision_time / time_horizon)
        danger_score *= (1.0 + time_factor)

    # 4. 考虑子弹速度：速度越快越危险
    bullet_speed = np.sqrt(bullet['vx']**2 + bullet['vy']**2)
    speed_factor = min(bullet_speed / 10.0, 1.0)  # 归一化
    danger_score *= (0.5 + 0.5 * speed_factor)

    return min(danger_score, 1.0)
```

#### 考虑子弹方向的优化版本

```python
def calculate_bullet_danger_with_direction(bullet, player_pos, player_velocity):
    """
    考虑子弹方向的危险度计算
    """
    # 1. 计算子弹到自机的向量
    dx = player_pos[0] - bullet['x']
    dy = player_pos[1] - bullet['y']
    distance = np.sqrt(dx * dx + dy * dy)

    if distance < 0.01:
        return 1.0  # 已经重叠

    # 2. 归一化方向向量
    to_player_dir = np.array([dx / distance, dy / distance])

    # 3. 子弹速度方向
    bullet_speed = np.sqrt(bullet['vx']**2 + bullet['vy']**2)
    if bullet_speed < 0.01:
        # 静止子弹
        bullet_dir = np.array([0, 0])
    else:
        bullet_dir = np.array([bullet['vx'] / bullet_speed,
                               bullet['vy'] / bullet_speed])

    # 4. 计算子弹方向与自机方向的夹角余弦值
    # cos(θ) = 1 表示正对着自机，-1表示背离自机
    direction_alignment = np.dot(bullet_dir, to_player_dir)

    # 5. 只有朝向自机的子弹才危险
    if direction_alignment < 0:
        # 子弹背离自机，危险度很低
        return 0.0

    # 6. 计算碰撞时间（假设直线运动）
    # 使用相对速度
    relative_vx = bullet['vx'] - player_velocity[0]
    relative_vy = bullet['vy'] - player_velocity[1]
    relative_speed = np.sqrt(relative_vx**2 + relative_vy**2)

    if relative_speed < 0.01:
        # 相对静止
        collision_time = float('inf')
    else:
        # 计算最近接近时间
        collision_time = distance / relative_speed

    # 7. 计算最近接近距离
    if collision_time > 0 and collision_time < 60:  # 60帧内
        # 预测位置
        future_bullet_x = bullet['x'] + bullet['vx'] * collision_time
        future_bullet_y = bullet['y'] + bullet['vy'] * collision_time
        future_player_x = player_pos[0] + player_velocity[0] * collision_time
        future_player_y = player_pos[1] + player_velocity[1] * collision_time

        closest_distance = np.sqrt(
            (future_bullet_x - future_player_x)**2 +
            (future_bullet_y - future_player_y)**2
        )

        # 考虑判定半径
        effective_distance = closest_distance - bullet['radius'] - PLAYER_HITBOX_RADIUS
    else:
        effective_distance = distance - bullet['radius'] - PLAYER_HITBOX_RADIUS

    # 8. 计算危险度
    danger = 0.0

    # 距离因素
    if effective_distance < 0:
        danger = 1.0
    elif effective_distance < 100:
        danger = 1.0 - (effective_distance / 100.0)

    # 方向因素：越对准自机越危险
    danger *= direction_alignment

    # 速度因素：越快越危险
    speed_factor = min(bullet_speed / 10.0, 1.0)
    danger *= (0.5 + 0.5 * speed_factor)

    # 时间因素：越快到达越危险
    if collision_time < 60:
        time_factor = 1.0 - (collision_time / 60.0)
        danger *= (0.5 + 0.5 * time_factor)

    # 大小因素：子弹越大越危险
    size_factor = min(bullet['radius'] / 10.0, 1.0)
    danger *= (0.7 + 0.3 * size_factor)

    return min(danger, 1.0)
```

### 方法2：8方向危险度评估

**核心思想**：评估自机向8个方向移动后的危险度

```python
def calculate_8direction_danger(bullets, player_pos, player_velocity):
    """
    计算8个方向的危险度

    Returns:
        danger_8dir: 8个方向的危险度列表
    """
    # 8个方向的单位向量
    directions = [
        (0, -1),    # 上
        (0, 1),     # 下
        (-1, 0),    # 左
        (1, 0),     # 右
        (-1, -1),   # 左上
        (1, -1),    # 右上
        (-1, 1),    # 左下
        (1, 1)      # 右下
    ]

    # 归一化对角线方向
    for i in range(4, 8):
        length = np.sqrt(2)
        directions[i] = (directions[i][0] / length,
                        directions[i][1] / length)

    danger_scores = []
    move_speed = 4.0  # 自机移动速度（像素/帧）

    for direction in directions:
        # 模拟向该方向移动
        test_velocity = (direction[0] * move_speed,
                        direction[1] * move_speed)

        # 计算该方向的总危险度
        total_danger = 0.0

        # 预测未来10帧的位置
        for t in range(1, 11):
            test_pos = (
                player_pos[0] + test_velocity[0] * t,
                player_pos[1] + test_velocity[1] * t
            )

            # 检查是否出界
            if (test_pos[0] < 0 or test_pos[0] > SCREEN_WIDTH or
                test_pos[1] < 0 or test_pos[1] > SCREEN_HEIGHT):
                total_danger += 1.0  # 出界很危险
                continue

            # 计算该位置的危险度
            frame_danger = 0.0
            for bullet in bullets:
                # 预测子弹在t帧后的位置
                bullet_future_x = bullet['x'] + bullet['vx'] * t
                bullet_future_y = bullet['y'] + bullet['vy'] * t

                # 计算距离
                dx = test_pos[0] - bullet_future_x
                dy = test_pos[1] - bullet_future_y
                distance = np.sqrt(dx * dx + dy * dy)

                # 考虑判定半径
                effective_distance = distance - bullet['radius'] - PLAYER_HITBOX_RADIUS

                if effective_distance < 0:
                    frame_danger += 1.0
                elif effective_distance < 30:
                    frame_danger += (1.0 - effective_distance / 30.0)

            # 时间衰减：越远的未来权重越低
            time_weight = 1.0 - (t / 10.0) * 0.5
            total_danger += frame_danger * time_weight

        # 归一化
        danger_scores.append(min(total_danger / 10.0, 1.0))

    return danger_scores
```

### 方法3：危险热力图

**核心思想**：生成整个游戏区域的危险度热力图

```python
def generate_danger_heatmap(bullets, grid_size=16):
    """
    生成危险度热力图

    Args:
        bullets: 子弹列表
        grid_size: 网格大小（像素）

    Returns:
        heatmap: 危险度热力图 (H, W)
    """
    height = SCREEN_HEIGHT // grid_size
    width = SCREEN_WIDTH // grid_size
    heatmap = np.zeros((height, width), dtype=np.float32)

    # 预测未来20帧
    time_horizon = 20

    for t in range(time_horizon):
        # 时间衰减权重
        time_weight = 1.0 - (t / time_horizon) * 0.7

        for bullet in bullets:
            # 预测子弹位置
            future_x = bullet['x'] + bullet['vx'] * t
            future_y = bullet['y'] + bullet['vy'] * t

            # 转换为网格坐标
            grid_x = int(future_x / grid_size)
            grid_y = int(future_y / grid_size)

            # 检查边界
            if grid_x < 0 or grid_x >= width or grid_y < 0 or grid_y >= height:
                continue

            # 计算影响范围（根据子弹大小）
            influence_radius = int(bullet['radius'] / grid_size) + 2

            for dy in range(-influence_radius, influence_radius + 1):
                for dx in range(-influence_radius, influence_radius + 1):
                    nx = grid_x + dx
                    ny = grid_y + dy

                    if nx < 0 or nx >= width or ny < 0 or ny >= height:
                        continue

                    # 计算距离
                    distance = np.sqrt(dx * dx + dy * dy) * grid_size

                    # 危险度随距离衰减
                    if distance < bullet['radius'] + PLAYER_HITBOX_RADIUS:
                        danger = 1.0
                    elif distance < bullet['radius'] + 50:
                        danger = 1.0 - ((distance - bullet['radius']) / 50.0)
                    else:
                        danger = 0.0

                    # 考虑子弹速度
                    bullet_speed = np.sqrt(bullet['vx']**2 + bullet['vy']**2)
                    speed_factor = min(bullet_speed / 10.0, 1.0)
                    danger *= (0.5 + 0.5 * speed_factor)

                    # 累加危险度
                    heatmap[ny, nx] += danger * time_weight

    # 归一化
    max_danger = heatmap.max()
    if max_danger > 0:
        heatmap = heatmap / max_danger

    return heatmap
```

## 综合危险度评估系统

```python
class DangerEvaluator:
    """危险度评估系统"""

    def __init__(self):
        self.player_hitbox_radius = 2.0  # 自机判定点半径
        self.time_horizon = 30  # 预测时间范围（帧）

    def evaluate(self, game_state):
        """
        综合评估危险度

        Returns:
            {
                'overall_danger': float,      # 整体危险度
                'direction_danger': list,     # 8方向危险度
                'nearest_threat': dict,       # 最近的威胁
                'safe_directions': list,      # 安全方向
                'heatmap': np.array          # 危险热力图（可选）
            }
        """
        bullets = game_state['bullets']
        player_pos = (game_state['player_x'], game_state['player_y'])
        player_vel = (game_state['player_vx'], game_state['player_vy'])

        # 1. 计算每个子弹的危险度
        bullet_dangers = []
        for bullet in bullets:
            danger = self.calculate_bullet_danger_with_direction(
                bullet, player_pos, player_vel
            )
            bullet_dangers.append({
                'bullet': bullet,
                'danger': danger
            })

        # 按危险度排序
        bullet_dangers.sort(key=lambda x: x['danger'], reverse=True)

        # 2. 整体危险度（考虑多个子弹的叠加）
        overall_danger = 0.0
        for i, bd in enumerate(bullet_dangers[:10]):  # 只考虑最危险的10个
            weight = 1.0 / (i + 1)  # 权重递减
            overall_danger += bd['danger'] * weight
        overall_danger = min(overall_danger, 1.0)

        # 3. 8方向危险度
        direction_danger = self.calculate_8direction_danger(
            bullets, player_pos, player_vel
        )

        # 4. 找出安全方向
        safe_threshold = 0.3
        safe_directions = [i for i, d in enumerate(direction_danger)
                          if d < safe_threshold]

        # 5. 最近的威胁
        nearest_threat = None
        if bullet_dangers:
            nearest_threat = {
                'bullet': bullet_dangers[0]['bullet'],
                'danger': bullet_dangers[0]['danger'],
                'distance': self._calculate_distance(
                    bullet_dangers[0]['bullet'], player_pos
                )
            }

        return {
            'overall_danger': overall_danger,
            'direction_danger': direction_danger,
            'nearest_threat': nearest_threat,
            'safe_directions': safe_directions
        }
```

```python
# 在环境中使用
class TouhouEnv(gym.Env):

    def _get_observation(self):

        # 构建特征向量
```

## 优化建议

### 1. 性能优化
```python
# 使用numpy向量化计算
def calculate_all_bullet_dangers_vectorized(bullets, player_pos, player_vel):
    """向量化计算所有子弹的危险度"""
    if not bullets:
        return []

    # 转换为numpy数组
    bullet_positions = np.array([[b['x'], b['y']] for b in bullets])
    bullet_velocities = np.array([[b['vx'], b['vy']] for b in bullets])
    bullet_radii = np.array([b['radius'] for b in bullets])

    # 向量化计算距离
    dx = player_pos[0] - bullet_positions[:, 0]
    dy = player_pos[1] - bullet_positions[:, 1]
    distances = np.sqrt(dx * dx + dy * dy)

    # 向量化计算方向对齐度
    to_player = np.stack([dx, dy], axis=1)
    to_player_norm = to_player / (distances[:, np.newaxis] + 1e-6)

    bullet_speeds = np.sqrt(bullet_velocities[:, 0]**2 + bullet_velocities[:, 1]**2)
    bullet_dirs = bullet_velocities / (bullet_speeds[:, np.newaxis] + 1e-6)

    direction_alignment = np.sum(bullet_dirs * to_player_norm, axis=1)

    # 计算危险度
    effective_distances = distances - bullet_radii - PLAYER_HITBOX_RADIUS
    dangers = np.where(
        effective_distances < 0,
        1.0,
        np.where(
            effective_distances < 100,
            1.0 - effective_distances / 100.0,
            0.0
        )
    )

    # 应用方向因素
    dangers *= np.maximum(direction_alignment, 0)

    return dangers
```

### 2. 缓存优化
```python
class DangerEvaluator:
    def __init__(self):
        self.cache = {}
        self.cache_lifetime = 5  # 缓存5帧

    def evaluate(self, game_state, frame_id):
        # 检查缓存
        if frame_id in self.cache:
            return self.cache[frame_id]

        # 计算
        result = self._compute_danger(game_state)

        # 更新缓存
        self.cache[frame_id] = result

        # 清理旧缓存
        old_frames = [f for f in self.cache if f < frame_id - self.cache_lifetime]
        for f in old_frames:
            del self.cache[f]

        return result
```

## 总结

危险度评估的关键要素：
1. **子弹轨迹预测**：考虑速度和加速度
2. **方向判断**：只有朝向自机的子弹才危险
3. **时间因素**：越快到达越危险
4. **大小因素**：子弹越大越危险
5. **多子弹叠加**：密集弹幕的综合威胁

推荐使用**方法2（8方向危险度）**作为特征输入，因为它：
- 维度固定（8维）
- 直接对应动作空间
- 计算效率高
- 信息丰富
